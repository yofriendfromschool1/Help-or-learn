local Players = game:GetService("Players") -- Players
local LocalPlayer = Players.LocalPlayer -- LocalPlayer
local Character = LocalPlayer.Character -- Character
local Humanoid = Character.Humanoid -- Humanoid
local Location = Character.HumanoidRootPart.CFrame -- Location of player
local function Speed(des) -- Changes LocalPlayer speed to desired choice
    Humanoid.WalkSpeed = des
end
local function Jump(des) -- Changes LocalPlayer jump power to desired choice
    Humanoid.JumpPower = des
end
local function Gravity(dess) -- Changes LocalPlayer Gravity to desired choice
    workspace.Gravity = dess
end
local function TeleportToCFrame(x,y,z) -- Teleports Player to cframe if yk what that is
    Character.HumanoidRootPart.CFrame = CFrame.new(x,y,z)
end
local function TeleportToPlayer(name) -- Teleports to player just put their name in in quotes so like 'yosad' or "dasdda"
    local lol
    local istt
    for i,v in pairs(Players:GetDescendants()) do
        if v.Name == name then
            istt = true
            lol = v.Character.HumanoidRootPart.CFrame
        end
    end
    if not istt then
        error("Could not find player MAKE SURE TO TYPE FULL USERNAME NOT DISPLAY NAME")
    end
    Character.HumanoidRootPart.CFrame = lol
end
local function TeleportToPart(pname) -- Teleports to part name
    local isdsa
    for i,v in pairs(workspace:GetDescendants()) do
        if v.Name == pname and v:IsA("Part") then
            isdsa = true
            Character.HumanoidRootPart.CFrame = v.CFrame
            break
        elseif v.Name == pname and v:IsA("BasePart") then
            isdsa = true
            Character.HumanoidRootPart.CFrame = v.CFrame
            break
        elseif v.Name == pname and v:IsA("MeshPart") then
            isdsa = true
            Character.HumanoidRootPart.CFrame = v.CFrame
            break
        end
    end
    if not isdsa then
        error("Could not find part lol")
    end
end
